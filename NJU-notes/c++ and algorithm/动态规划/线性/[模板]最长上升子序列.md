O(n^2)
```C++
#include<bits/stdc++.h>
#define MAX 5005
using namespace std;
int a[MAX],d[MAX];
int dp(int n) {
	d[1] = 1;
	int ans = 1;
	for (int i = 2;i <= n;i++) 
	{
		d[i] = 1;
		for (int j = 1;j < i;j++) {
			if (a[j] < a[i]) {
				d[i] = max(d[j] + 1, d[i]);
			}
		}
		ans = max(ans, d[i]);
	}
	return ans;
}
int main() {
	ios::sync_with_stdio(false);
	cin.tie(0);
	int n;
	cin >> n;
	for (int i = 1;i <= n;i++)  cin >> a[i];
	int ans=dp(n);
	cout << ans;
	return 0;
}
```

O(nlogn)
```C++
int n, a[MAXN], d[MAXN], di[MAXN], pre[MAXN], res[MAXN];

int dp() {
    int ans = 0; // 当前找到的最长子序列的长度，也代表 d 数组当前的有效长度

    for (int i = 1; i <= n; ++i) {
        // 【核心步骤 1：二分查找】
        // 在 d 数组里找第一个比 a[i] 大的位置。
        // upper_bound 是为了处理“不下降”（允许相等），如果是“严格上升”要用 lower_bound
        // tmp 表示 a[i] 应该放到 d 数组的哪个位置（下标）
        // 也就是说，a[i] 将成为长度为 tmp+1 的子序列的新结尾
        int tmp = std::upper_bound(d, d + ans, a[i]) - d;

        // 【核心步骤 2：记录族谱（这是最难懂的一句）】
        // 如果 tmp > 0，说明 a[i] 接在了一个长度为 tmp 的序列后面。
        // 那个长度为 tmp 的序列的结尾是谁呢？是 d[tmp-1]。
        // 但我们需要的是那个结尾在原数组 a 里的下标，所以去查 di[tmp-1]。
        // 如果 tmp == 0，说明 a[i] 是头，没有前驱，记为 -1。
        pre[i] = tmp ? di[tmp - 1] : -1;

        // 【核心步骤 3：更新潜力榜】
        // 贪心策略：既然 a[i] 更小（或能延长），就用 a[i] 更新 d[tmp]
        d[tmp] = a[i];

        // 【核心步骤 4：记录影子下标】
        // 记下来：现在 d[tmp] 这个位置存的值，对应的是原数组的第 i 个元素
        di[tmp] = i; 

        // 【核心步骤 5：更新最大长度】
        // 如果 tmp 刚好等于当前的长度 ans，说明我们开辟了新的长度，ans 加 1
        if (tmp == ans) ++ans;
    }

    // 【核心步骤 6：顺藤摸瓜还原序列】
    // 倒着推：
    // 1. 找到最长序列的最后一个元素。它的位置存在 di[ans-1] 里。
    // 2. 把值存入 res[k]。
    // 3. 通过 pre[i] 找到它的上家，跳过去，继续循环。
    for (int k = ans, i = di[ans - 1]; k; --k) {
        res[k] = a[i]; // 存下当前数字
        i = pre[i];    // 跳到“上家”的位置
    }

    return ans; // 返回最长长度
}
```
