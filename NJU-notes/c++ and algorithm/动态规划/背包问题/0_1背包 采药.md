# 0/1背包 采药

[https://www.luogu.com.cn/problem/P1048#ide](https://www.luogu.com.cn/problem/P1048#ide)  
原地dp  
注意点：正序 or 逆序？  
逆序才可保证用到的数据均为旧数据而没有重复采药。  
```cpp
#include <bits/stdc++.h>
#define T 1005
#define M 105
int d[T];
int times[M], value[M];
using namespace std;
int main() {
	ios::sync_with_stdio(false);
	cin.tie(0);
	int t, m;
	cin >> t >> m;
	for (int i = 1; i <= m; i++) cin >> times[i] >> value[i];
	for (int i = 1;i <= m;i++) {
		for (int j = t;j >=times[i];j--) {
			d[j] = max(d[j - times[i]] + value[i], d[j]);
		}
	}
	cout<<d[t];
	return 0;
}
```

```cpp
for(int i=1;i<=n;i++)
```

一维数组优化：

```cpp
for(int i=1;i<=n;i++)
{
    for(int c=m;c>=0;c--)
    {
        if(c>=w[i])
        f[c]=max(f[c],f[c-w[i]]+v[i]);
    }
}
```

更进一步的常数优化：

```cpp
for(int i=1;i<=n;i++)
{
    sumw+=w[i];
    bound=max(m-sumw,w[i]);
    for(int c=m;c>=bound;c--)
    {
        if(c>=w[i])
        f[c]=max(f[c],f[c-w[i]]+v[i]);
    }
}
```

多重背包问题：

```cpp
for(int i=1;i<=n;i++)
{
    if(w[i]*a[i]>m)
    {
        for(int c=0;c<=m;c++)
        {
        if(c>=w[i])
        f[c]=max(f[c],f[c-w[i]]+v[i]);
        }
    }
    else
    {
         k=1;amount=a[i];
         while(k<amount)
         {
             for(int c=k*w[i];c>=0;c--)
             {
                 if(c>=w[i])
                 f[c]=max(f[c],f[c-w[i]]+k*v[i]);
             }
             amount-=k;
             k<<=1;
         }  
         for(int c=amount*w[i];c>=0;c--)
         {
             f[c]=max(f[c],f[c-w[i]]+amount*v[i]);
         }
    } 
}
```
