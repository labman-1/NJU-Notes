注意点：正序 or 逆序？  
逆序才可保证用到的数据均为旧数据而没有重复采药。  
```cpp
#include <bits/stdc++.h>
#define T 1005
#define M 105
int d[T];
int times[M], value[M];
using namespace std;
int main() {
	ios::sync_with_stdio(false);
	cin.tie(0);
	int t, m;
	cin >> t >> m;
	for (int i = 1; i <= m; i++) cin >> times[i] >> value[i];
	for (int i = 1;i <= m;i++) {
		for (int j = t;j >=times[i];j--) {
			d[j] = max(d[j - times[i]] + value[i], d[j]);
		}
	}
	cout<<d[t];
	return 0;
}
```

一维数组优化：

```cpp
for(int i=1;i<=n;i++)
{
    for(int c=m;c>=0;c--)
    {
        if(c>=w[i])
        f[c]=max(f[c],f[c-w[i]]+v[i]);
    }
}
```

更进一步的常数优化：
这个优化**极其危险**！它有一个致命前提：**你必须只关心f\[m\] 的最终结果！** 如果题目要求你输出“不超过容量 m 的最大价值”（即需要在 f\[0\]…f\[m\] 里取个最大值），这个优化会直接导致小容量的结果全部算错。
```cpp
// 1. 预处理【后缀和】(suffix_sum)
// suffix_sum[i] 表示从第 i 个物品到最后一个物品的重量总和
int suffix_sum[30];
suffix_sum[n+1] = 0;
for (int i = n; i >= 1; i--) {
    suffix_sum[i] = suffix_sum[i+1] + w[i];
}

// 2. 正确的界限优化
for(int i = 1; i <= n; i++) {
    // 关键点：用总容量 减去 【当前物品及之后所有物品的总重量】
    int bound = max(w[i], m - suffix_sum[i]); 
    
    for(int j = m; j >= bound; j--) {
        f[j] = max(f[j], f[j - w[i]] + v[i]);
    }
}
```

[P1064 [NOIP 2006 提高组] 金明的预算方案 - 洛谷](https://www.luogu.com.cn/problem/P1064)
```c++
#include<bits/stdc++.h>
using namespace std;
int n,m;
constexpr int MAX=200005;
struct item{
	int v,p,q;
	bool operator<(const item& other){
		return q<other.q;
	}
	bool operator!(){
		return (v==0)&&(p==0)&&(q==0);
	}
};
int dp[MAX];
vector<item>things[3];
int main(){
	ios::sync_with_stdio(false);cin.tie(0);
	cin>>n>>m;
	int size=0;
	for(int i=0;i<=2;i++){
		things[i].resize(m+1);
	}
	for(int i = 1;i<=m;i++){
		int v,p,q;
		cin>>v>>p>>q;
			things[0][i]={v,p,q};
			if(q!=0&&!things[1][q]){
				things[1][q]={v,p,q};
			}
			else if(q!=0){
				things[2][q]={v,p,q};
			}
	}
	for(int i = 1;i<=m;i++){
		item t=things[0][i];
		if(t.q==0){	
			item t1=things[1][i];
		item t2=things[2][i];
		for(int j = n;j>=t.v;j--){	
	     dp[j]=max(dp[j],dp[j-t.v]+t.v*t.p);
	     if(j-t.v-t1.v>=0){
	     	dp[j]=max(dp[j],dp[j-t.v-t1.v]+t.v*t.p+t1.v*t1.p);
		 }
		   if(j-t.v-t2.v>=0){
		   	dp[j]=max(dp[j],dp[j-t.v-t2.v]+t.v*t.p+t2.v*t2.p);
		   }
		 if(j-t.v-t1.v-t2.v>=0){
		 	 dp[j]=max(dp[j],dp[j-t.v-t1.v-t2.v]+t.v*t.p+t1.p*t1.v+t2.p*t2.v);  
		 } 
}
}
}
cout<<dp[n];
	return 0;
}
```
