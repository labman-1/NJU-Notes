# Video Game Troubles
本题显然是背包dp的变种，存在依赖关系，无法直接套用模板。考虑将游戏与游戏机打包在一起，对于某一个游戏机而言，购买其附属的游戏的方案是互斥的，所以“找到某个游戏机对应的最佳方案”本身就是一个较小的、单独的dp问题。所以主dp遍历游戏机，在主dp内部顺序进行了三次遍历，第一次是标记钱够不够买游戏机（这是子dp的前提条件，解决依赖问题），第二次是进行子dp找到某个游戏机的最佳方案并据此更新相应的dp值，第三次遍历v并将子dp存在的沙盒更新到主dp中。评估复杂度为O(NGV)，可以通过。
https://www.luogu.com.cn/problem/P2967
```c++
#include<bits/stdc++.h>
using namespace std;
int n,v;
constexpr int MAX=100005;
vector<pair<int,int>>mGame[51];
vector<int>machi;
long long dp[MAX];
long long temp[MAX];
int main(){
	ios::sync_with_stdio(false);cin.tie(0);
	cin>>n>>v;
	machi.resize(n);
	for(int i = 0;i<n;i++){
		int p,g;
		cin>>p>>g;
		machi[i]=p;
		for(int j= 0;j<g;j++){
			int gp,pv;
			cin>>gp>>pv;
			mGame[i].push_back({gp,pv});
		}
	}
	for(int i = 0;i<n;i++){
		for(int j = 0;j<=v;j++){
			if(j>=machi[i]){
				temp[j]=dp[j-machi[i]];
			}
			else{
				temp[j]=-1;
			}
		}
		for(auto g:mGame[i]){
			for(int j = v;j>=g.first+machi[i];j--){
				if(temp[j-g.first]!=-1){
					temp[j]=max(temp[j],temp[j-g.first]+1ll*g.second);
				}
			}
		}
		for(int j = 0;j<=v;j++){
			dp[j]=max(dp[j],temp[j]);
		}
	}
	long long ans=0;
	for(int i = 1;i<=v;i++){
		ans=max(ans,dp[i]);
	}
	cout<<ans;
	return 0;
}

```
