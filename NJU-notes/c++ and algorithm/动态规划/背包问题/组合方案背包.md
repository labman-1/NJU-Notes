# Video Game Troubles
https://www.luogu.com.cn/problem/P2967
```c++
#include<bits/stdc++.h>
using namespace std;
int n,v;
constexpr int MAX=100005;
vector<pair<int,int>>mGame[51];
vector<int>machi;
long long dp[MAX];
long long temp[MAX];
int main(){
	ios::sync_with_stdio(false);cin.tie(0);
	cin>>n>>v;
	machi.resize(n);
	for(int i = 0;i<n;i++){
		int p,g;
		cin>>p>>g;
		machi[i]=p;
		for(int j= 0;j<g;j++){
			int gp,pv;
			cin>>gp>>pv;
			mGame[i].push_back({gp,pv});
		}
	}
	for(int i = 0;i<n;i++){
		for(int j = 0;j<=v;j++){
			if(j>=machi[i]){
				temp[j]=dp[j-machi[i]];
			}
			else{
				temp[j]=-1;
			}
		}
		for(auto g:mGame[i]){
			for(int j = v;j>=g.first+machi[i];j--){
				if(temp[j-g.first]!=-1){
					temp[j]=max(temp[j],temp[j-g.first]+1ll*g.second);
				}
			}
		}
		for(int j = 0;j<=v;j++){
			dp[j]=max(dp[j],temp[j]);
		}
	}
	long long ans=0;
	for(int i = 1;i<=v;i++){
		ans=max(ans,dp[i]);
	}
	cout<<ans;
	return 0;
}
```
