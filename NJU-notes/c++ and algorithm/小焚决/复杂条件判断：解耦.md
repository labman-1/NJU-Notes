#include<bits/stdc++.h>
#define N 105
using namespace std;
int n,curr;
int main() {
	ios::sync_with_stdio(false);cin.tie(0);
	cin >> n;
	bool is_first = true;
	for (int i = n;i >= 0;i--) {
		cin >> curr;
		if (curr == 0) continue;
		if (!is_first && curr > 0)cout << "+";
		if (curr < 0)cout << "-";
		if (abs(curr) != 1 || i == 0) cout << abs(curr);
		if (i > 1) cout << "x^" << i;
		else if (i == 1) cout << "x";
		is_first = false;
	}
	
	return  0;
}
