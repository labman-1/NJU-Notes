#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

const int N = 1e5 + 5;
int raw[N];          // 原始数据
vector<int> vals;    // 待离散化容器

int main() {
    int n; cin >> n;
    for (int i = 1; i <= n; i++) {
        cin >> raw[i];
        vals.push_back(raw[i]);
    }

    // 1. 排序
    sort(vals.begin(), vals.end());
    // 2. 去重
    vals.erase(unique(vals.begin(), vals.end()), vals.end());

    // 3. 使用：将原始值转换为下标（从 0 或 1 开始）
    for (int i = 1; i <= n; i++) {
        // 从 1 开始的映射（推荐，方便树状数组）
        int id = lower_bound(vals.begin(), vals.end(), raw[i]) - vals.begin() + 1;
        // 此时 id ∈ [1, vals.size()]
    }

    // 反向查询（如果需要还原）
    // int origin = vals[id - 1];
    return 0;
}
