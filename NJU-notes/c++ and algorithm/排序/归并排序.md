[P1309 [NOIP 2011 普及组] 瑞士轮 - 洛谷](https://www.luogu.com.cn/problem/P1309)
刚开始我的思路是直接模拟比赛，每次两两对决，然后重新排序，这样的复杂度为O(R\*nlogn)。注意到每次对决分为winner/loser，且它们内部的相对顺序不变，这与归并排序的思想很像，由此优化为O(N)，使用双指针归并。
```c++
#include<bits/stdc++.h>
#define MAX 200005
using namespace std;
int N, R, Q;
struct part {
	int score,strength,num;
};
bool partGreater(const part& a, const part& b) {
	if (a.score == b.score) {
		return a.num < b.num;
	}
	return a.score > b.score;
}
part par[MAX];
part winner[MAX], loser[MAX];
int main() {
	ios::sync_with_stdio(false);cin.tie(0);
	cin >> N >> R >> Q;
	int total = N * 2;
	for (int i = 1;i <= total;i++) {
		cin >> par[i].score;
		par[i].num = i;
	}
	for (int i = 1;i <= total;i++) {
		cin >> par[i].strength;
	}
	sort(par+1, par+total+1, partGreater);
	for (int i = 1;i <= R;i++) {
		int win_index = 0,lose_index = 0;
		for (int j = 1;j <= total;j += 2) {
			if (par[j].strength > par[j + 1].strength) {
				par[j].score++;
				winner[++win_index] = par[j];
				loser[++lose_index] = par[j + 1];
			}
			else {
				par[j + 1].score++;
				winner[++win_index] = par[j+1];
				loser[++lose_index] = par[j];
			}
		}
		int wi = 1, li = 1,pari=1;
		while (wi <= win_index && li <= lose_index) {
			if (partGreater(winner[wi], loser[li])) {
				par[pari++] = winner[wi++];
			}
			else {
				par[pari++] = loser[li++];
			}
		}
		while (wi <= win_index ) {
			par[pari++] = winner[wi++];
		}
		while ( li <= lose_index) {
			par[pari++] = loser[li++];
		}
	}
	cout << par[Q].num;
	return 0;
}
```
