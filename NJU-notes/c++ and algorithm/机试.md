using namespace std;
int t, n;
constexpr int N = 10005;
constexpr long long INF = 0x3f3f3f3f3f3f3f3f;
long long dp[10005][2]; // 第一维比n的上界大一些即可
// dp[i][0]表示前i层搞定，第i层为A；dp[i][1]表示第i层为B。
long long charge[N];  // A
long long repair[N];  // B
long long prefixB[N]; // 0到i层均为B站，i+1层为A时，前i层总代价
long long suffixB[N], prefixA[N], suffixA[N];
// 返回一段纯A区间的cost(只需计算B类即可)
long long costA(int s, int e) {
	if(s==0)
            return prefixA[e];
	else if (e==n-1)
            return suffixA[s];
    long long ans = 0;
    for (int i = s; i <= e; i++) {
        ans += min(i - s + 1, e - i + 1) * repair[i];
    }
    return ans;
}
long long costB(int s, int e) {
    long long ans = 0;
	if(s==0)
            return prefixB[e];
	else if (e==n-1)
            return suffixB[s];
    for (int i = s; i <= e; i++) {
        ans += min(i - s + 1, e - i + 1) * charge[i];
    }
    return ans;
}
void solve() {
    cin >> n;
    memset(dp, INF, sizeof(dp));
    memset(prefixA, 0, sizeof(prefixA));
    memset(prefixB, 0, sizeof(prefixB));
    memset(suffixA, 0, sizeof(suffixA));
    memset(suffixB, 0, sizeof(suffixB));
    for (int i = 0; i < n; i++) {
        cin >> charge[i] >> repair[i];
    }
    for (int i = 0; i < n; i++) {
        for (int j = 0; j <= i; j++) {
            prefixB[i] += (i - j+1) * charge[j];
            prefixA[i] += (i - j+1) * repair[j];
        }
    }
    for (int i = n - 1; i >= 0; i--) {
        for (int j = n - 1; j >= i; j--) {
            suffixB[i] += (j - i+1) * charge[j];
            suffixA[i] += (j - i+1) * repair[j];
        }
    }
    for (int i = 0; i <= n - 2;i++){
dp[i][0] = costA(0, i);
    dp[i][1] = costB(0, i);
	}
        
    for (int i = 1; i <= n - 1; i++) {
        for (int j = 1; j <= i; j++) {
            dp[i][0] = min(dp[i][0], dp[j - 1][1] + costA(j, i));
            dp[i][1] = min(dp[i][1], dp[j - 1][0] + costB(j, i));
        }
    }
    long long ans = min(dp[n - 1][0], dp[n - 1][1]);
    cout << ans << '\n';
}
int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}

struct dot{
	int weight,id;
	dot(int a,int b):weight(a),id(b){
	}
const	bool operator<(const dot&other)const {
	return weight<other.weight;
}
};
vector<dot>d;
void floyd(){
	for(auto i:d){
			for(auto j : d){
				ans[i.id][j.id]=dist[i.id][j.id]+max({i.weight,j.weight});
			}
		}
	for(auto k:d){
		for(auto i:d){
			for(auto j : d){
			if(dist[i.id][j.id]>dist[i.id][k.id]+dist[k.id][j.id]){
				dist[i.id][j.id]=dist[i.id][k.id]+dist[k.id][j.id];
				ans[i.id][j.id]=min(ans[i.id][j.id],dist[i.id][j.id]+max({i.weight,j.weight,k.weight}));
			}
			}
		}
	}
}

#include<bits/stdc++.h>
using namespace std;
constexpr int M=1e7;
int n,m;
class Fenwick{
	private:
		int size;
		vector<long long>c;
	public:
		Fenwick(int s):size(s),c(s+1,0){
		}
	int lowbit(int x){
		return x&-x;
	}
	long long query_range(int s,int e){
		return query(e)-query(s-1);
	}
	long long query(int x){
		long long sum=0;
		while(x>0){
			sum+=c[x];
			x-=lowbit(x);
		}
		return sum;
	}
	void update(int x,int k){
		while(x<=size){
			c[x]+=k;
			x+=lowbit(x);
		}
	}
};
int main(){
	ios::sync_with_stdio(false);cin.tie(0);
	cin>>n>>m;
	vector<int>a(n+1);
	vector<int>prevB(n+1,0);
	for(int i= 1;i<=n;i++){
		cin>>a[i];
	}
	Fenwick f(m);
	long long ans=0;
	for(int i = 1;i<=m;i++){
		int b;
		cin>>b;
		if(!prevB[b]){
			prevB[b]=i;
			f.update(prevB[b],a[b]);
			ans+=f.query(i-1);
		}
		else{
			f.update(prevB[b],-a[b]);
			f.update(i,a[b]);
			ans+=f.query_range(prevB[b]+1,i-1);
			prevB[b]=i;
		}
	}
	cout<<ans;
	return 0;
}

[P1091 [NOIP 2004 提高组] 合唱队形 - 洛谷](https://www.luogu.com.cn/problem/P1091)
![[机试_202606031500.png]]

我们当时通过分析，把这道题从复杂的“背包 DP”直接降维成了**“前缀乘积累加”**。我们来快速复盘它的核心物理逻辑：

---

### 一、 核心物理约束与组合分解

我们要拼出形如 $T_1 < T_2 < \dots < T_i > T_{i+1} > \dots > T_K$ 的山峰序列（因为 $K$ 只是定义，我们要求的是**所有可能长度**的队形总数） [INDEX]。

我们将所有人按身高从小到大排序并分组，得到 $m$ 个不同的身高：$H_1 < H_2 < \dots < H_m$，对应的频次分别为 $c_1, c_2, \dots, c_m$ [INDEX]。

现在，我们**强制让身高 $H_i$ 担任这整个队形的最高点（Peak）** [INDEX]。

1.  **最高点本身 $H_i$**：必须且只能选 1 个人。从 $c_i$ 个人里挑 1 个，方案数是 **$c_i$** [INDEX]。
2.  **矮于最高点的任意身高 $H_x$（$x < i$）**：
    由于它比最高点矮，且两侧必须严格单调，所以**这个身高在整个队形里最多只能选 2 个人**（左边最多1个，右边最多1个）。我们面临三种互斥的选择：
    *   **选 0 个人**（不放）：$1$ 种方案。
    *   **选 1 个人**（放左边或右边）：从 $c_x$ 人里挑 1 个（$c_x$ 种），放左或放右（2 种），方案数 = **$2c_x$** [INDEX]。
    *   **选 2 个人**（一左一右）：从 $c_x$ 人里挑 2 个排列。方案数 = **$c_x(c_x - 1)$** [INDEX]。
    
    **该身高的总贡献因子：**
    $$1 + 2c_x + c_x(c_x - 1) = \mathbf{c_x^2 + c_x + 1}$$ [INDEX]

3.  **高于最高点的任何身高（$x > i$）**：绝对不能选（方案数为 0）。

---

### 二、 终极数学公式：前缀乘积

因为每一个矮身高的选择在逻辑上是**完全独立、互不干扰**的（因为只要选出来，高度天然决定了它们在左右两翼的唯一顺序） [INDEX]。

所以，以 $H_i$ 为最高点的总方案数，就是**所有比它矮的身高因子的累乘**，再乘上它自己的选择数：
$$\text{Ways}(H_i) = c_i \times \prod_{x=1}^{i-1} (c_x^2 + c_x + 1)$$ [INDEX]

全局总答案，就是枚举每一个身高作为最高点时的方案数之和 [INDEX]：
$$\text{Total\_Ans} = \sum_{i=1}^{m} \left( c_i \times \prod_{x=1}^{i-1} (c_x^2 + c_x + 1) \right)$$ [INDEX]

---

### 三、 极简代码模板（肌肉记忆）

我们在写代码时，甚至**不需要开辟任何 DP 数组**！只需要在从小到大遍历身高的过程中，顺手用一个变量维护 **`currProduct`（前缀积）** 即可：

```cpp
long long ans = 0;
long long currProduct = 1; // 维护所有比当前矮的身高的方案数累乘

for (const auto& g : groups) {
    long long c = g.count;

    // 1. 当前身高作为最高点，结算它的总贡献
    long long currAns = (currProduct * c) % MOD;
    ans = (ans + currAns) % MOD;

    // 2. 更新前缀乘积，把当前身高作为“矮身高”加入累乘，供后面更高的人使用
    long long factor = (c * c + c + 1) % MOD;
    currProduct = (currProduct * factor) % MOD;
}
```

整个算法只在最开始需要一次 `sort` 排序，其余部分是纯粹的 **$O(N)$ 线性扫描**。

#include<bits/stdc++.h>
using namespace std;
int n;
vector<int>a;
int main(){
	ios::sync_with_stdio(false);cin.tie(0);
	cin>>n;
	a.assign(n,0);
	int sum=0;
	for(int i = 0;i<n;i++){
		cin>>a[i];
		sum+=a[i];
	}
	int pos=0;
	bool lastdir=0;//0=left,1=right
	while(sum--){
		if(!pos){
			a[pos++]--;
			cout<<'R';
			lastdir=1;
		}
		else if(pos==n){
			a[--pos]--;
			cout<<'L';
			lastdir=0;
			if(!a[pos]){
				n--;
			}
		}
		else{
			//之前向右走，现在倾向于保持方向不变，除非右边走不了
			if(lastdir){
				if(a[pos]){
					cout<<'R';
					a[pos++]--;
				}
				else{
					cout<<'L';
					a[pos--]--;
					lastdir=0;
				}
			}
			else{
				if(a[pos-1]==1&&a[pos]){
					cout<<'R';
					a[pos++]--;
					lastdir=1;
				}
				else{
					cout<<'L';
					a[--pos]--;
				}
			}
		}
	}
	return 0;
}
