```c++
#include<bits/stdc++.h>
using namespace std;
constexpr int M=1e7;
int n,m;
class Fenwick{
	private:
		int size;
		vector<long long>c;
	public:
		Fenwick(int s):size(s),c(s+1,0){
		}
	int lowbit(int x){
		return x&-x;
	}
	long long query_range(int s,int e){
		return query(e)-query(s-1);
	}
	long long query(int x){
		long long sum=0;
		while(x>0){
			sum+=c[x];
			x-=lowbit(x);
		}
		return sum;
	}
	void update(int x,int k){
		while(x<=size){
			c[x]+=k;
			x+=lowbit(x);
		}
	}
};
int main(){
	ios::sync_with_stdio(false);cin.tie(0);
	cin>>n>>m;
	vector<int>a(n+1);
	vector<int>prevB(n+1,0);
	for(int i= 1;i<=n;i++){
		cin>>a[i];
	}
	Fenwick f(m);
	long long ans=0;
	for(int i = 1;i<=m;i++){
		int b;
		cin>>b;
		if(!prevB[b]){
			prevB[b]=i;
			f.update(prevB[b],a[b]);
			ans+=f.query(i-1);
		}
		else{
			f.update(prevB[b],-a[b]);
			f.update(i,a[b]);
			ans+=f.query_range(prevB[b]+1,i-1);
			prevB[b]=i;
		}
	}
	cout<<ans;
	return 0;
}
```
