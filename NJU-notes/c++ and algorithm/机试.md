```c++
#include <bits/stdc++.h>
using namespace std;
int t, n;
constexpr int N = 10005;
constexpr long long INF = 0x3f3f3f3f3f3f3f3f;
long long dp[10005][2]; // 第一维比n的上界大一些即可
// dp[i][0]表示前i层搞定，第i层为A；dp[i][1]表示第i层为B。
long long charge[N];  // A
long long repair[N];  // B
long long prefixB[N]; // 0到i层均为B站，i+1层为A时，前i层总代价
long long suffixB[N], prefixA[N], suffixA[N];
// 返回一段纯A区间的cost(只需计算B类即可)
long long costA(int s, int e) {
	if(s==0)
            return prefixA[e];
	else if (e==n-1)
            return suffixA[s];
    long long ans = 0;
    for (int i = s; i <= e; i++) {
        ans += min(i - s + 1, e - i + 1) * repair[i];
    }
    return ans;
}
long long costB(int s, int e) {
    long long ans = 0;
	if(s==0)
            return prefixB[e];
	else if (e==n-1)
            return suffixB[s];
    for (int i = s; i <= e; i++) {
        ans += min(i - s + 1, e - i + 1) * charge[i];
    }
    return ans;
}
void solve() {
    cin >> n;
    memset(dp, INF, sizeof(dp));
    memset(prefixA, 0, sizeof(prefixA));
    memset(prefixB, 0, sizeof(prefixB));
    memset(suffixA, 0, sizeof(suffixA));
    memset(suffixB, 0, sizeof(suffixB));
    for (int i = 0; i < n; i++) {
        cin >> charge[i] >> repair[i];
    }
    for (int i = 0; i < n; i++) {
        for (int j = 0; j <= i; j++) {
            prefixB[i] += (i - j+1) * charge[j];
            prefixA[i] += (i - j+1) * repair[j];
        }
    }
    for (int i = n - 1; i >= 0; i--) {
        for (int j = n - 1; j >= i; j--) {
            suffixB[i] += (j - i+1) * charge[j];
            suffixA[i] += (j - i+1) * repair[j];
        }
    }
    for (int i = 0; i <= n - 2;i++){
dp[i][0] = costA(0, i);
    dp[i][1] = costB(0, i);
	}
        
    for (int i = 1; i <= n - 1; i++) {
        for (int j = 1; j <= i; j++) {
            dp[i][0] = min(dp[i][0], dp[j - 1][1] + costA(j, i));
            dp[i][1] = min(dp[i][1], dp[j - 1][0] + costB(j, i));
        }
    }
    long long ans = min(dp[n - 1][0], dp[n - 1][1]);
    cout << ans << '\n';
}
int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}
```

```c++
#include <bits/stdc++.h>
using namespace std;
int t, n;
constexpr int N = 10005;
constexpr long long INF = 0x3f3f3f3f3f3f3f3f;
long long dp[10005][2]; // 第一维比n的上界大一些即可
// dp[i][0]表示前i层搞定，第i层为A；dp[i][1]表示第i层为B。
long long charge[N];  // A
long long repair[N];  // B
long long sumkR[N],sumkC[N],sumR[N],sumC[N];
//return the cost of the area that [s,e] are A(charge)
long long costA(int s, int e) {
	if(s==0) return (e+1)*sumR[e]-sumkR[e]; 
	if(e==n-1)return sumkR[e]-sumkR[s-1]-(s-1)*(sumR[e]-sumR[s-1]);
	int mid = s+((e-s)>>1);
    return  (e+1)*(sumR[e]-sumR[mid])+(1-s)*(sumR[mid]-sumR[s-1])+sumkR[mid]-sumkR[s-1]-sumkR[e]+sumkR[mid];
}
long long costB(int s, int e) {
	if(s==0) return (e+1)*sumC[e]-sumkC[e]; 
	if(e==n-1)return sumkC[e]-sumkC[s-1]-(s-1)*(sumC[e]-sumC[s-1]);
	int mid = s+((e-s)>>1);
    return  (e+1)*(sumC[e]-sumC[mid])+(1-s)*(sumC[mid]-sumC[s-1])+sumkC[mid]-sumkC[s-1]-sumkC[e]+sumkC[mid];
}
void solve() {
    cin >> n;
    memset(dp, INF, sizeof(dp));
    memset(sumkC, 0, sizeof(sumkC));
    memset(sumkR, 0, sizeof(sumkR));
     memset(sumC, 0, sizeof(sumC));
    memset(sumR, 0, sizeof(sumR));
    for (int i = 0; i < n; i++) {
        cin >> charge[i] >> repair[i];
    }
    for (int i = 0; i <n; i++) {
         long long prevR = (i == 0) ? 0 : sumR[i-1];
    	long long prevC = (i == 0) ? 0 : sumC[i-1];
    	long long prevkR = (i == 0) ? 0 : sumkR[i-1];
   	 long long prevkC = (i == 0) ? 0 : sumkC[i-1];

    sumR[i] = prevR + repair[i];
    sumC[i] = prevC + charge[i];
    sumkR[i] = prevkR + 1LL * i * repair[i]; // 严格乘以下标 i
    sumkC[i] = prevkC + 1LL * i * charge[i]; // 1LL 防止 int 乘法溢出
    }

    for (int i = 0; i <= n - 2;i++){
dp[i][0] = costA(0, i);
    dp[i][1] = costB(0, i);
	}
        
    for (int i = 1; i <= n - 1; i++) {
        for (int j = 1; j <= i; j++) {
            dp[i][0] = min(dp[i][0], dp[j - 1][1] + costA(j, i));
            dp[i][1] = min(dp[i][1], dp[j - 1][0] + costB(j, i));
        }
    }
    long long ans = min(dp[n - 1][0], dp[n - 1][1]);
    cout << ans << '\n';
}
int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}
```

```c++
#include<bits/stdc++.h>
using namespace std;
constexpr int M=1e7;
int n,m;
class Fenwick{
	private:
		int size;
		vector<long long>c;
	public:
		Fenwick(int s):size(s),c(s+1,0){
		}
	int lowbit(int x){
		return x&-x;
	}
	long long query_range(int s,int e){
		return query(e)-query(s-1);
	}
	long long query(int x){
		long long sum=0;
		while(x>0){
			sum+=c[x];
			x-=lowbit(x);
		}
		return sum;
	}
	void update(int x,int k){
		while(x<=size){
			c[x]+=k;
			x+=lowbit(x);
		}
	}
};
int main(){
	ios::sync_with_stdio(false);cin.tie(0);
	cin>>n>>m;
	vector<int>a(n+1);
	vector<int>prevB(n+1,0);
	for(int i= 1;i<=n;i++){
		cin>>a[i];
	}
	Fenwick f(m);
	long long ans=0;
	for(int i = 1;i<=m;i++){
		int b;
		cin>>b;
		if(!prevB[b]){
			prevB[b]=i;
			f.update(prevB[b],a[b]);
			ans+=f.query(i-1);
		}
		else{
			f.update(prevB[b],-a[b]);
			f.update(i,a[b]);
			ans+=f.query_range(prevB[b]+1,i-1);
			prevB[b]=i;
		}
	}
	cout<<ans;
	return 0;
}
```

```c++
#include<bits/stdc++.h>
using namespace std;
constexpr int MAX=2000005;
int a[MAX];
int n;
int prefix[31][MAX];
int target[MAX];
//return costs of operation in [s,e]
int execute(int start,int end,int x){
	return end-start+1-(prefix[x][end]-prefix[x][start-1]);//start-1 might be 0, it's OK
}
//main solve method 
int solve(int start,int end,int x){
	int mid =start+((end-start)>>1);
	if(start==end){
		return execute(start,end,x);
	}
	int a=solve(start,mid,x+1)+execute(mid+1,end,x);
	int b=execute(start,mid,x)+solve(mid+1,end,x+1);
	//b<=a -> execute left half->left is determined
	if(b<=a){
		fill(target+start,target+mid+1,x);
		return b;
	}
	else{
		fill(target+mid+1,target+end+1,x);
		return a;
	}
}


int main(){
	ios::sync_with_stdio(false);cin.tie(0);
	cin>>n;
	for(int i = 1;i<=n;i++){
		cin>>a[i];
	}
	
	//initialize prefix 
	for(int i = 1;i<=30;i++){
		int sum=0;
		for(int j = 1;j<=n;j++){
		if(a[j]==i){
			sum++;
		}
		prefix[i][j]=sum;
	}
	}
	cout<<solve(1,n,1)<<'\n';
	for(int i=1;i<=n;i++){
		cout<<target[i]<<' ';
	}
	
	return 0;
}
```
