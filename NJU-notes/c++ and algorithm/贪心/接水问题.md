### 总结  

1注意归并排序（迭代版）的边界条件和合并过程  
21\. 初始化：分配一个与原数组 x同样大小的临时数组 tmp。 2\. 分段循环：设定一个步长 seg，从1开始，每次翻倍（seg \= 2），直到 seg >= len。这模拟了递归分解的过程。 3\. 分组合并：在每次步长下，遍历整个数组，每次处理长度为 seg2的一个区间。 确定当前区间的起始 start、中点 mid (start + seg)和结束 end (min(start + seg\*2, len))。 对两个有序子数组 \[start, mid)和 \[mid, end)进行合并 4\. 指针交换：每一轮合并完成后，交换源数组和临时数组的角色，为下一轮更大的合并做准备 5\. 结果确认：所有合并完成后，确认有序的结果在哪个数组中，并确保 x正确指向它。（记得delete，资源管理要谨慎）  
3先用高精度/整数类型完成所有精确计算，最后再进行一次可能引入误差的浮点运算  
```c++
#include<bits/stdc++.h>
using namespace std;
typedef struct people {
	int num;
	int tcost;
}p;
bool operator<=(const people& x,const people& y) {
	return x.tcost <= y.tcost;
}
void merge_sort(p*& x, int len) {
	p* a = x;
	p* b = new p[len];
	for (int seg = 1;seg < len;seg=seg << 1) {
		for (int start = 0;start < len;start += (seg << 1)) {
			int low = start, high = min(start + seg + seg, len), mid = min(start + seg, len);
			int start1 = low, start2 = mid, end1 = mid, end2 = high;
			while (start1 < end1 && start2 < end2) {
				if (a[start1] <= a[start2])b[low++] = a[start1++];
				else b[low++] = a[start2++];
			}
			while (start1 < end1) b[low++] = a[start1++];
			while (start2 < end2) b[low++] = a[start2++];
		}
		p* temp = a;
		a = b;
		b = temp;
	}
	if (a != x) {
		for (int i = 0;i < len;i++) x[i] = a[i];
		delete[]a;
	}
	else delete[]b;
}
int main() {
	int n;
	cin >> n;
	p* x = new p[n];
	for (int i = 0;i < n;i++) {
		x[i].num = i+1;
		cin >> x[i].tcost;
	} 
	merge_sort(x,n);
	long long sum_time = 0;
	for (int i = 0;i < n;i++) {
		cout << x[i].num<<" ";
		sum_time += (long long)(n-i-1)*x[i].tcost;
	}
	double average_time = (double)sum_time / n;
	cout << endl;
	cout << fixed<<setprecision(2)<<average_time;
	delete[]x;
	return 0;
}
```
