## [洛谷P1080国王游戏](https://www.luogu.com.cn/problem/P1080#ide)  
#高精度+ #贪心 （邻项交换法）

```cpp
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef struct person {
	int l, r;
	bool operator<(const person& x)const {
		return max(l * r, x.r) < max(r, x.l * x.r);
	}
}*m;
struct big_int {
	vector<int>digits;
	big_int(){}
	big_int(ll x) {
		while (x > 0) {
			digits.push_back(x % 10);
			x /= 10;
		}
		if (digits.empty()) {
			digits.push_back(0);
		}
	}
	big_int operator *(ll x) {
		big_int res;
		ll carry=0;
		res.digits.assign(digits.size() + 20, 0);
		for (int i = 0;i < digits.size() || carry;i++) {
			if (i < digits.size()) {
				carry += (ll)digits[i] * x;
			}
			res.digits[i]=carry % 10;
			carry /= 10;
		}
		while (res.digits.size() > 1 && res.digits.back() == 0) {
			res.digits.pop_back();
		}
		return res;
	}
	big_int operator /(ll x) {
		big_int res;
		res.digits.assign(digits.size(), 0);
		ll remainder = 0;
		for (int i = digits.size() - 1;i >= 0;i--) {
			remainder = remainder * 10 + digits[i];
			res.digits[i] = remainder / x;
			remainder %= x;
		}
		while (res.digits.size() > 1 && res.digits.back() == 0) {
			res.digits.pop_back();
		}
		return res;
	}
	string to_string() const{
		string s;
		for (int i = digits.size() - 1;i >= 0;i--) {
			s += char('0' + digits[i]);
		}
		return s;
	}
	bool operator >(const big_int& other) const{
		if (digits.size() != other.digits.size()) return digits.size() > other.digits.size();
		for (int i = digits.size() - 1;i >= 0;i--) {
			if (digits[i] != other.digits[i]) return digits[i] > other.digits[i];
		}
		return false;
	}
	bool operator <( const big_int& other) const{
		return other > *this;
	}
};
int main() {
	int n, a, b;
	cin >> n;
	cin >> a >> b;
	vector<person>mi(n + 1);
	mi[0].l = a, mi[0].r = b;
	for (int i = 1;i <= n;i++) {
		cin >> a >> b;
		mi[i].l = a, mi[i].r = b;
	}
	sort(mi.begin()+1, mi.end());
	big_int ans(0);
	big_int prefix_product(1);
	for (int i = 1; i <= n; i++) {
		prefix_product = prefix_product* mi[i-1].l;
		ans = max(ans, prefix_product/mi[i].r);
	}
	cout << ans.to_string() << endl;
	return 0;
}
```
