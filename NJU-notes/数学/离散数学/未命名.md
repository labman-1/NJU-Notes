
$$\begin{aligned}
\text{左式} & \equiv \neg(p \wedge q) \vee (\neg p \wedge q) \\
& \equiv (\neg p \vee \neg q) \vee (\neg p \wedge q) \quad && (\text{德·摩根律}) \\
& \equiv \neg q \vee \big(\neg p \vee (\neg p \wedge q)\big) \quad && (\text{结合律、交换律}) \\
& \equiv \neg q \vee \neg p \quad && (\text{吸收律}) \\
& \equiv \neg p \vee \neg q \quad && (\text{交换律}) \\
& \equiv \text{右式}
\end{aligned}$$

$$\begin{aligned}
\text{左式} & \equiv \neg(p \wedge q) \vee (\neg p \wedge q) \\
& \equiv (\neg p \vee \neg q) \vee (\neg p \wedge q) \quad && (\text{德·摩根律}) \\
& \equiv \big((\neg p \vee \neg q) \vee \neg p\big) \wedge \big((\neg p \vee \neg q) \vee q\big) \quad && (\text{分配律}) \\
& \equiv \big((\neg p \vee \neg p) \vee \neg q\big) \wedge \big(\neg p \vee (\neg q \vee q)\big) \quad && (\text{结合律、交换律}) \\
& \equiv (\neg p \vee \neg q) \wedge (\neg p \vee \text{T}) \quad && (\text{幂等律、排中律}) \\
& \equiv (\neg p \vee \neg q) \wedge \text{T} \quad && (\text{同一律/支配律}) \\
& \equiv \neg p \vee \neg q \quad && (\text{同一律}) \\
& \equiv \text{右式}
\end{aligned}$$
