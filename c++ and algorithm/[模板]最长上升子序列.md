```
#include<bits/stdc++.h>
#define MAX 5005
using namespace std;
int a[MAX],d[MAX];
int dp(int n) {
	d[1] = 1;
	int ans = 1;
	for (int i = 2;i <= n;i++) 
	{
		d[i] = 1;
		for (int j = 1;j < i;j++) {
			if (a[j] < a[i]) {
				d[i] = max(d[j] + 1, d[i]);
			}
		}
		ans = max(ans, d[i]);
	}
	return ans;
}
int main() {
	ios::sync_with_stdio(false);
	cin.tie(0);
	int n;
	cin >> n;
	for (int i = 1;i <= n;i++)  cin >> a[i];
	int ans=dp(n);
	cout << ans;
	return 0;
}
```
