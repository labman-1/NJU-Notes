# 奖学金

partial\_sort  
```c++
#include <bits/stdc++.h>
using namespace std;

struct score {
	int Chinese, math, English, num, total;
};

// 加上 const，保证安全且符合标准
bool cmp(const score &a, const score &b) {
	if (a.total != b.total) return a.total > b.total;
	else {
		if (a.Chinese != b.Chinese) return a.Chinese > b.Chinese;
		else return a.num < b.num;
	}
}

int main() {
	ios::sync_with_stdio(false);
	cin.tie(0);
	int n;
	cin >> n;
	vector<score> stu(n);
	for (int i = 0; i < n; i++) {
		stu[i].num = i + 1;
		cin >> stu[i].Chinese >> stu[i].math >> stu[i].English;
		stu[i].total = stu[i].Chinese + stu[i].English + stu[i].math;
	}

    // 严谨的处理：取 n 和 5 的较小值，防止越界
    int k = min(n, 5);
	partial_sort(stu.begin(), stu.begin() + k, stu.end(), cmp);

	for (int i = 0; i < k; i++) { // 这里直接用 k 就行了
		cout << stu[i].num << ' ' << stu[i].total << '\n';
	}
	return 0;
}
```
