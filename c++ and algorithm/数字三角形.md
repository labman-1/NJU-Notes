# 数字三角形

[https://www.luogu.com.cn/problem/P1216#ide](https://www.luogu.com.cn/problem/P1216#ide)  

```cpp
#include<bits/stdc++.h>
#define MAX 1005
using namespace std;
int d[MAX];
int a[MAX][MAX];
int r;
void dp() {
	for (int i = 1;i <= r;i++) d[i] = a[r][i];
	for (int i = r - 1;i > 0;i--) {
		for (int j = 1;j <=i;j++) {
			d[j] =a[i][j] +max(d[j], d[j + 1]);
		}
	}
}
int main() {
	ios::sync_with_stdio(false);
	cin.tie(0);
	cin >> r;
	for (int i = 1;i <= r;i++)  
		for (int j = 1;j <= i;j++) cin >> a[i][j];
	dp();
	cout << d[1];
	return 0;
}
```
